async function loadDashboardStats() { const token = localStorage.getItem("token"); if (!token) { window.location.href = "/index.html"; return; } const response = await fetch(`${API_URL}/files`, { headers: { "Authorization": `Bearer ${token}` } }); const files = await response.json(); let total = files.length; let high = 0; let medium = 0; let safe = 0; const recentUploads = document.getElementById("recentUploads"); if (recentUploads) recentUploads.innerHTML = ""; files.forEach(file => { const filename = file.filename || "Unknown file"; const risk = analyzeRisk(filename); if (risk === "HIGH") high++; else if (risk === "MEDIUM") medium++; else safe++; if (recentUploads) { recentUploads.innerHTML += ` ${escapeHtml(filename)} ${risk} Encrypted & Stored 🔐 AES-256 `; } }); document.getElementById("totalFiles").textContent = total; document.getElementById("highRisk").textContent = high; document.getElementById("mediumRisk").textContent = medium; document.getElementById("safeFiles").textContent = safe; const storagePercent = Math.min(total * 12, 100); document.getElementById("storageFill").style.width = storagePercent + "%"; document.getElementById("storageText").textContent = storagePercent + "% used"; document.getElementById("storagePercent").textContent = storagePercent + "%"; const score = Math.max(45, 100 - high * 18 - medium * 8); document.getElementById("threatScore").textContent = score; if (total === 0 && recentUploads) { recentUploads.innerHTML = `No uploaded files found.`; } } if (window.location.pathname.includes("dashboard.html")) { const token = localStorage.getItem("token"); const userEmail = localStorage.getItem("userEmail"); if (!token) { window.location.href = "/index.html"; } else { const emailEl = document.getElementById("userEmail"); if (emailEl) emailEl.textContent = userEmail || "Kullanıcı"; loadDashboardStats(); setTimeout(() => { addSecurityAlert("🛡️ Cloud security monitoring system activated"); }, 700); } }
